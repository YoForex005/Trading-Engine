**Hook 1: useChartData (OHLC data fetching)**
```typescript
// clients/desktop/src/features/trading/hooks/useChartData.ts
import { useState, useEffect } from 'react';
import type { OHLCData } from '../types';

export function useChartData(symbol: string, timeframe: string) {
  const [ohlc, setOhlc] = useState<OHLCData[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<Error | null>(null);

  useEffect(() => {
    let cancelled = false;

    async function fetchOHLC() {
      try {
        const res = await fetch(`/api/ohlc/${symbol}/${timeframe}`);
        if (!res.ok) throw new Error('Failed to fetch OHLC data');

        const data = await res.json();
        if (!cancelled) {
          setOhlc(data);
          setLoading(false);
        }
      } catch (err) {
        if (!cancelled) {
          setError(err as Error);
          setLoading(false);
        }
      }
    }

    fetchOHLC();

    return () => {
      cancelled = true;
    };
  }, [symbol, timeframe]);

  return { ohlc, loading, error };
}
```

**Hook 2: useTradingWebSocket (real-time price updates)**
```typescript
// clients/desktop/src/features/trading/hooks/useTradingWebSocket.ts
import { useState, useEffect } from 'react';
import type { TickData } from '../types';

export function useTradingWebSocket(symbol: string) {
  const [lastTick, setLastTick] = useState<TickData | null>(null);
  const [status, setStatus] = useState<'connecting' | 'connected' | 'disconnected'>('connecting');

  useEffect(() => {
    const ws = new WebSocket(`ws://localhost:8080/market-data`);

    ws.onopen = () => {
      setStatus('connected');
      ws.send(JSON.stringify({ action: 'subscribe', symbol }));
    };

    ws.onmessage = (event) => {
      const tick = JSON.parse(event.data) as TickData;
      if (tick.symbol === symbol) {
        setLastTick(tick);
      }
    };

    ws.onerror = () => setStatus('disconnected');
    ws.onclose = () => setStatus('disconnected');

    return () => {
      ws.close();
    };
  }, [symbol]);

  return { lastTick, status };
}
```

import { useMemo } from 'react';
import type { OHLCData, Indicator } from '../types';
import { calculateSMA, calculateEMA, calculateRSI } from '@/indicators/core';

export function useIndicators(ohlc: OHLCData[], activeIndicators: string[]) {
  const indicators = useMemo(() => {
    const result: Record<string, Indicator> = {};

    if (activeIndicators.includes('SMA_20')) {
      result.SMA_20 = { values: calculateSMA(ohlc, 20) };
    }

    if (activeIndicators.includes('EMA_50')) {
      result.EMA_50 = { values: calculateEMA(ohlc, 50) };
    }

    if (activeIndicators.includes('RSI_14')) {
      result.RSI_14 = { values: calculateRSI(ohlc, 14) };
    }

    return result;
  }, [ohlc, activeIndicators]);

  return indicators;
}

import type { Indicator } from '../types';

type Props = {
  indicators: Record<string, Indicator>;
  activeIndicators: string[];
  onToggleIndicator: (indicators: string[]) => void;
};

export function IndicatorPane({ indicators, activeIndicators, onToggleIndicator }: Props) {
  const availableIndicators = ['SMA_20', 'EMA_50', 'RSI_14', 'MACD'];

  const handleToggle = (indicator: string) => {
    if (activeIndicators.includes(indicator)) {
      onToggleIndicator(activeIndicators.filter((i) => i !== indicator));
    } else {
      onToggleIndicator([...activeIndicators, indicator]);
    }
  };

  return (
    <div className="indicator-pane">
      <h3>Indicators</h3>
      <div className="indicator-list">
        {availableIndicators.map((indicator) => (
          <label key={indicator}>
            <input
              type="checkbox"
              checked={activeIndicators.includes(indicator)}
              onChange={() => handleToggle(indicator)}
            />
            {indicator}
          </label>
        ))}
      </div>

      <div className="indicator-values">
        {Object.entries(indicators).map(([name, data]) => (
          <div key={name} className="indicator-value">
            {name}: {data.values[data.values.length - 1]?.toFixed(2)}
          </div>
        ))}
      </div>
    </div>
  );
}

### Task 5: Create Shared Custom Hooks

**What:** Extract reusable hooks into shared/hooks/

**Why:** Avoid duplication, DRY principle

**Details:**

**Shared hook: useWebSocket (generic WebSocket connection)**
```typescript
// clients/desktop/src/shared/hooks/useWebSocket.ts
import { useState, useEffect } from 'react';

type WebSocketStatus = 'connecting' | 'connected' | 'disconnected';

export function useWebSocket<T>(url: string) {
  const [data, setData] = useState<T | null>(null);
  const [status, setStatus] = useState<WebSocketStatus>('connecting');

  useEffect(() => {
    const ws = new WebSocket(url);

    ws.onopen = () => setStatus('connected');
    ws.onmessage = (event) => setData(JSON.parse(event.data));
    ws.onerror = () => setStatus('disconnected');
    ws.onclose = () => setStatus('disconnected');

    return () => ws.close();
  }, [url]);

  return { data, status };
}
```

**Shared hook: useFetch (generic API fetching)**
```typescript
// clients/desktop/src/shared/hooks/useFetch.ts
import { useState, useEffect } from 'react';

export function useFetch<T>(url: string) {
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<Error | null>(null);

  useEffect(() => {
    let cancelled = false;

    async function fetchData() {
      try {
        const res = await fetch(url);
        if (!res.ok) throw new Error(`HTTP ${res.status}`);

        const json = await res.json();
        if (!cancelled) {
          setData(json);
          setLoading(false);
        }
      } catch (err) {
        if (!cancelled) {
          setError(err as Error);
          setLoading(false);
        }
      }
    }

    fetchData();

    return () => {
      cancelled = true;
    };
  }, [url]);

  return { data, loading, error };
}
```

## Anti-Patterns to Avoid

- **Over-splitting:** Don't create a component for every 10 lines (aim for 50-400 lines per component)
- **Prop drilling hell:** Use context or state management for deeply nested props (>3 levels)
- **Premature abstraction:** Extract hooks when logic is repeated, not preemptively
- **Breaking tests:** Run tests after every refactor step to catch regressions early
