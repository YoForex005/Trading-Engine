package e2e

import (
	"context"

// TestE2E_OrderExecution verifies complete order flow:
// 1. Create account
// 2. Place market order
// 3. Verify order executed
// 4. Verify position created
// 5. Verify balance updated
// 6. Verify database persistence
func TestE2E_OrderExecution(t *testing.T) {
	// Setup test database connection
	ctx := context.Background()
	pool, err := database.Connect(ctx, getDatabaseURL(t))
	if err != nil {
		t.Fatalf("database connect failed: %v", err)
	}
	defer pool.Close()

	// Clean up test data
	defer cleanupTestData(t, pool)

	// Create repositories
	accountRepo := repository.NewAccountRepository(pool)
	orderRepo := repository.NewOrderRepository(pool)
	positionRepo := repository.NewPositionRepository(pool)
	tradeRepo := repository.NewTradeRepository(pool)

	// Step 1: Create test account
	accountNumber := "E2E_TEST_001"
	initialBalance := testutil.MustParseDecimal(t, "10000.00")

	accountID, err := accountRepo.Create(ctx, accountNumber, initialBalance, "USD", 100)
	if err != nil {
		t.Fatalf("failed to create account: %v", err)
	}

	// Step 2: Place market buy order
	order := &bbook.Order{
		AccountID: accountID,
		Symbol:    "EURUSD",
		Type:      "market",
		Side:      "buy",
		Lots:      testutil.MustParseDecimal(t, "1.0"),
	}

	currentPrice := testutil.MustParseDecimal(t, "1.0850")
	engine := bbook.NewEngine()

	executed, err := engine.ExecuteOrder(ctx, order, currentPrice)
	if err != nil {
		t.Fatalf("order execution failed: %v", err)
	}

	if !executed {
		t.Fatal("order should have executed but didn't")
	}

	// Step 3: Verify order persisted to database
	dbOrder, err := orderRepo.GetByID(ctx, order.ID)
	if err != nil {
		t.Fatalf("failed to fetch order: %v", err)
	}

	if dbOrder.Status != "filled" {
		t.Errorf("order status: got %s, want filled", dbOrder.Status)
	}

	// Step 4: Verify position created
	positions, err := positionRepo.GetByAccount(ctx, accountID)
	if err != nil {
		t.Fatalf("failed to fetch positions: %v", err)
	}

	if len(positions) != 1 {
		t.Fatalf("expected 1 position, got %d", len(positions))
	}

	position := positions[0]
	if position.Symbol != "EURUSD" {
		t.Errorf("position symbol: got %s, want EURUSD", position.Symbol)
	}

	expectedLots := testutil.MustParseDecimal(t, "1.0")
	testutil.AssertDecimalEqual(t, position.Lots, expectedLots, "position lots")

	// Step 5: Verify account balance updated (margin deducted)
	account, err := accountRepo.GetByID(ctx, accountID)
	if err != nil {
		t.Fatalf("failed to fetch account: %v", err)
	}

	// Balance should have margin deducted (1.0 lot * 1.0850 price / 100 leverage = 1085.00 margin)
	expectedMargin := testutil.MustParseDecimal(t, "1085.00")
	testutil.AssertDecimalEqual(t, account.Margin, expectedMargin, "account margin")

	// Step 6: Verify trade recorded
	trades, err := tradeRepo.GetByAccount(ctx, accountID, 100, 0)
	if err != nil {
		t.Fatalf("failed to fetch trades: %v", err)
	}

	if len(trades) != 1 {
		t.Fatalf("expected 1 trade, got %d", len(trades))
	}

	trade := trades[0]
	testutil.AssertDecimalEqual(t, trade.EntryPrice, currentPrice, "trade entry price")
}

// TestE2E_PositionClose verifies position closing flow:
// 1. Open position
// 2. Close position
// 3. Verify position removed
// 4. Verify profit/loss calculated
// 5. Verify balance updated
func TestE2E_PositionClose(t *testing.T) {
	ctx := context.Background()
	pool, err := database.Connect(ctx, getDatabaseURL(t))
	if err != nil {
		t.Fatalf("database connect failed: %v", err)
	}
	defer pool.Close()
	defer cleanupTestData(t, pool)

	accountRepo := repository.NewAccountRepository(pool)
	positionRepo := repository.NewPositionRepository(pool)

	// Create account with open position
	accountNumber := "E2E_TEST_002"
	initialBalance := testutil.MustParseDecimal(t, "10000.00")
	accountID, _ := accountRepo.Create(ctx, accountNumber, initialBalance, "USD", 100)

	// Open position (buy EURUSD at 1.0850)
	entryPrice := testutil.MustParseDecimal(t, "1.0850")
	lots := testutil.MustParseDecimal(t, "1.0")
	positionID, _ := positionRepo.Create(ctx, accountID, "EURUSD", "buy", lots, entryPrice)

	// Close position at higher price (profit scenario)
	exitPrice := testutil.MustParseDecimal(t, "1.0900")
	engine := bbook.NewEngine()

	profit, err := engine.ClosePosition(ctx, positionID, exitPrice)
	if err != nil {
		t.Fatalf("close position failed: %v", err)
	}

	// Verify profit calculated correctly (50 pips * $10/pip = $500)
	expectedProfit := testutil.MustParseDecimal(t, "500.00")
	testutil.AssertDecimalEqual(t, profit, expectedProfit, "position profit")

	// Verify position removed from database
	position, err := positionRepo.GetByID(ctx, positionID)
	if err == nil {
		t.Error("position should be deleted after close")
	}

	// Verify account balance updated with profit
	account, _ := accountRepo.GetByID(ctx, accountID)
	expectedBalance := testutil.MustParseDecimal(t, "10500.00") // 10000 + 500 profit
	testutil.AssertDecimalEqual(t, account.Balance, expectedBalance, "balance after close")
}

// Helper functions

	// Use test database (not production)

}

func cleanupTestData(t *testing.T, pool *database.Pool) {
	ctx := context.Background()
	// Delete test accounts and related data
	_, err := pool.Exec(ctx, "DELETE FROM accounts WHERE account_number LIKE 'E2E_TEST_%'")
	if err != nil {
		t.Logf("cleanup warning: %v", err)
	}
}

import (

// TestE2E_PositionModify verifies modifying open positions:
// 1. Open position
// 2. Modify stop-loss
// 3. Modify take-profit
// 4. Verify modifications persisted
func TestE2E_PositionModify(t *testing.T) {
	ctx := context.Background()
	pool, err := database.Connect(ctx, getDatabaseURL(t))
	if err != nil {
		t.Fatalf("database connect failed: %v", err)
	}
	defer pool.Close()
	defer cleanupTestData(t, pool)

	accountRepo := repository.NewAccountRepository(pool)
	positionRepo := repository.NewPositionRepository(pool)

	// Create account and position
	accountID, _ := accountRepo.Create(ctx, "E2E_TEST_003", testutil.MustParseDecimal(t, "10000.00"), "USD", 100)
	entryPrice := testutil.MustParseDecimal(t, "1.0850")
	lots := testutil.MustParseDecimal(t, "1.0")
	positionID, _ := positionRepo.Create(ctx, accountID, "EURUSD", "buy", lots, entryPrice)

	engine := bbook.NewEngine()

	// Modify stop-loss
	stopLoss := testutil.MustParseDecimal(t, "1.0800")
	err = engine.SetStopLoss(ctx, positionID, stopLoss)
	if err != nil {
		t.Fatalf("set stop-loss failed: %v", err)
	}

	// Modify take-profit
	takeProfit := testutil.MustParseDecimal(t, "1.0950")
	err = engine.SetTakeProfit(ctx, positionID, takeProfit)
	if err != nil {
		t.Fatalf("set take-profit failed: %v", err)
	}

	// Verify modifications persisted
	position, err := positionRepo.GetByID(ctx, positionID)
	if err != nil {
		t.Fatalf("failed to fetch position: %v", err)
	}

	testutil.AssertDecimalEqual(t, position.StopLoss, stopLoss, "stop-loss")
	testutil.AssertDecimalEqual(t, position.TakeProfit, takeProfit, "take-profit")
}

// TestE2E_MultiplePositions verifies managing multiple concurrent positions:
// 1. Open multiple positions on different symbols
// 2. Verify all positions tracked independently
// 3. Close one position
// 4. Verify other positions unaffected
func TestE2E_MultiplePositions(t *testing.T) {
	ctx := context.Background()
	pool, err := database.Connect(ctx, getDatabaseURL(t))
	if err != nil {
		t.Fatalf("database connect failed: %v", err)
	}
	defer pool.Close()
	defer cleanupTestData(t, pool)

	accountRepo := repository.NewAccountRepository(pool)
	positionRepo := repository.NewPositionRepository(pool)

	// Create account
	accountID, _ := accountRepo.Create(ctx, "E2E_TEST_004", testutil.MustParseDecimal(t, "20000.00"), "USD", 100)

	// Open three positions
	lots := testutil.MustParseDecimal(t, "1.0")
	pos1ID, _ := positionRepo.Create(ctx, accountID, "EURUSD", "buy", lots, testutil.MustParseDecimal(t, "1.0850"))
	pos2ID, _ := positionRepo.Create(ctx, accountID, "GBPUSD", "buy", lots, testutil.MustParseDecimal(t, "1.2650"))
	pos3ID, _ := positionRepo.Create(ctx, accountID, "USDJPY", "sell", lots, testutil.MustParseDecimal(t, "148.50"))

	// Verify all three positions exist
	positions, _ := positionRepo.GetByAccount(ctx, accountID)
	if len(positions) != 3 {
		t.Fatalf("expected 3 positions, got %d", len(positions))
	}

	// Close GBPUSD position
	engine := bbook.NewEngine()
	_, err = engine.ClosePosition(ctx, pos2ID, testutil.MustParseDecimal(t, "1.2700"))
	if err != nil {
		t.Fatalf("close position failed: %v", err)
	}

	// Verify only EURUSD and USDJPY remain
	positions, _ = positionRepo.GetByAccount(ctx, accountID)
	if len(positions) != 2 {
		t.Fatalf("expected 2 positions after close, got %d", len(positions))
	}

	// Verify correct positions remain
	symbols := []string{positions[0].Symbol, positions[1].Symbol}
	if !contains(symbols, "EURUSD") || !contains(symbols, "USDJPY") {
		t.Error("wrong positions remain after close")
	}
}

func contains(slice []string, item string) bool {
	for _, s := range slice {
		if s == item {
			return true
		}
	}
	return false
}

# Run migrations
cd ../../
migrate -database "postgres://$DB_USER@localhost/$DB_NAME?sslmode=disable" -path db/migrations up

echo "Test database ready: $DB_NAME"
