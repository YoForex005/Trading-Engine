import { describe, it, expect, vi } from 'vitest'
import { screen, waitFor } from '@testing-library/react'
import { renderWithProviders, userEvent } from '../../test/utils'
import { IndicatorManager } from './IndicatorManager'

describe('IndicatorManager', () => {
  it('renders add indicator button', () => {
    renderWithProviders(<IndicatorManager onIndicatorChange={vi.fn()} />)

    const addButton = screen.getByRole('button', { name: /add indicator/i })
    expect(addButton).toBeInTheDocument()
  })

  it('opens indicator selection menu on button click', async () => {
    const user = userEvent.setup()
    renderWithProviders(<IndicatorManager onIndicatorChange={vi.fn()} />)

    const addButton = screen.getByRole('button', { name: /add indicator/i })
    await user.click(addButton)

    // Menu should appear with indicator options
    await waitFor(() => {
      expect(screen.getByText(/Moving Average/i)).toBeInTheDocument()
    })
  })

  it('adds indicator when user selects from menu', async () => {
    const user = userEvent.setup()
    const onIndicatorChange = vi.fn()

    renderWithProviders(<IndicatorManager onIndicatorChange={onIndicatorChange} />)

    // Open menu
    const addButton = screen.getByRole('button', { name: /add indicator/i })
    await user.click(addButton)

    // Select Moving Average
    const maOption = await screen.findByText(/Moving Average/i)
    await user.click(maOption)

    // Verify callback called with indicator data
    expect(onIndicatorChange).toHaveBeenCalledWith(
      expect.arrayContaining([
        expect.objectContaining({ type: 'MA' })
      ])
    )

    // Verify indicator appears in list
    await waitFor(() => {
      expect(screen.getByText(/MA\(/i)).toBeInTheDocument()
    })
  })

  it('removes indicator when user clicks remove button', async () => {
    const user = userEvent.setup()
    const onIndicatorChange = vi.fn()

    renderWithProviders(<IndicatorManager onIndicatorChange={onIndicatorChange} />)

    // First add an indicator
    const addButton = screen.getByRole('button', { name: /add indicator/i })
    await user.click(addButton)
    const maOption = await screen.findByText(/Moving Average/i)
    await user.click(maOption)

    // Now remove it
    const removeButton = await screen.findByRole('button', { name: /remove.*MA/i })
    await user.click(removeButton)

    // Verify indicator removed from list
    await waitFor(() => {
      expect(screen.queryByText(/MA\(/i)).not.toBeInTheDocument()
    })

    // Verify callback called with empty array
    expect(onIndicatorChange).toHaveBeenLastCalledWith([])
  })

  it('allows configuring indicator parameters', async () => {
    const user = userEvent.setup()
    const onIndicatorChange = vi.fn()

    renderWithProviders(<IndicatorManager onIndicatorChange={onIndicatorChange} />)

    // Add indicator
    const addButton = screen.getByRole('button', { name: /add indicator/i })
    await user.click(addButton)
    const maOption = await screen.findByText(/Moving Average/i)
    await user.click(maOption)

    // Find and click settings button
    const settingsButton = await screen.findByRole('button', { name: /settings.*MA/i })
    await user.click(settingsButton)

    // Change period from 14 to 20
    const periodInput = await screen.findByLabelText(/period/i)
    await user.clear(periodInput)
    await user.type(periodInput, '20')

    // Save changes
    const saveButton = screen.getByRole('button', { name: /save/i })
    await user.click(saveButton)

    // Verify callback called with updated period
    expect(onIndicatorChange).toHaveBeenLastCalledWith(
      expect.arrayContaining([
        expect.objectContaining({ type: 'MA', period: 20 })
      ])
    )
  })
})

import { describe, it, expect } from 'vitest'
import { screen } from '@testing-library/react'
import { renderWithProviders } from '../test/utils'
import { ErrorBoundary } from './ErrorBoundary'

// Component that throws an error for testing
const ThrowError = ({ shouldThrow }: { shouldThrow: boolean }) => {
  if (shouldThrow) {
    throw new Error('Test error')
  }
  return <div>Working component</div>
}

describe('ErrorBoundary', () => {
  it('renders children when no error occurs', () => {
    renderWithProviders(
      <ErrorBoundary>
        <ThrowError shouldThrow={false} />
      </ErrorBoundary>
    )

    expect(screen.getByText(/Working component/i)).toBeInTheDocument()
  })

  it('catches errors and displays error message', () => {
    // Suppress console.error for this test
    const consoleSpy = vi.spyOn(console, 'error').mockImplementation(() => {})

    renderWithProviders(
      <ErrorBoundary>
        <ThrowError shouldThrow={true} />
      </ErrorBoundary>
    )

    // Should show error UI instead of crashed component
    expect(screen.getByText(/something went wrong/i)).toBeInTheDocument()

    consoleSpy.mockRestore()
  })

  it('shows reload button when error occurs', () => {
    const consoleSpy = vi.spyOn(console, 'error').mockImplementation(() => {})

    renderWithProviders(
      <ErrorBoundary>
        <ThrowError shouldThrow={true} />
      </ErrorBoundary>
    )

    const reloadButton = screen.getByRole('button', { name: /reload/i })
    expect(reloadButton).toBeInTheDocument()

    consoleSpy.mockRestore()
