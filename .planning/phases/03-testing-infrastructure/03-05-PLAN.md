func TestLPManager_QuoteAggregation(t *testing.T) {
	tests := []struct {
		name          string
		symbol        string
		quotes        []Quote // Multiple LP quotes
		expectedBid   string  // Best bid (highest)
		expectedAsk   string  // Best ask (lowest)
	}{
		{
			name:   "single LP quote",
			symbol: "EURUSD",
			quotes: []Quote{
				{Symbol: "EURUSD", Bid: testutil.MustParseDecimal(t, "1.0850"), Ask: testutil.MustParseDecimal(t, "1.0852"), LP: "OANDA"},
			},
			expectedBid: "1.0850",
			expectedAsk: "1.0852",
		},
		{
			name:   "multiple LP quotes - pick best bid/ask",
			symbol: "EURUSD",
			quotes: []Quote{
				{Symbol: "EURUSD", Bid: testutil.MustParseDecimal(t, "1.0850"), Ask: testutil.MustParseDecimal(t, "1.0852"), LP: "OANDA"},
				{Symbol: "EURUSD", Bid: testutil.MustParseDecimal(t, "1.0851"), Ask: testutil.MustParseDecimal(t, "1.0853"), LP: "Binance"},
			},
			expectedBid: "1.0851", // Higher bid from Binance
			expectedAsk: "1.0852", // Lower ask from OANDA
		},
		{
			name:   "stale quote filtered out",
			symbol: "EURUSD",
			quotes: []Quote{
				{Symbol: "EURUSD", Bid: testutil.MustParseDecimal(t, "1.0850"), Ask: testutil.MustParseDecimal(t, "1.0852"), LP: "OANDA", Timestamp: time.Now()},
				{Symbol: "EURUSD", Bid: testutil.MustParseDecimal(t, "1.0900"), Ask: testutil.MustParseDecimal(t, "1.0902"), LP: "Stale", Timestamp: time.Now().Add(-10 * time.Minute)},
			},
			expectedBid: "1.0850", // Stale quote ignored
			expectedAsk: "1.0852",
		},
	}

	for _, tt := range tests {
		tt := tt
		t.Run(tt.name, func(t *testing.T) {
			t.Parallel()

			manager := NewLPManager()

			// Feed quotes to manager
			for _, quote := range tt.quotes {
				manager.ProcessQuote(quote)
			}

			// Get aggregated quote
			aggregated := manager.GetBestQuote(tt.symbol)

			expectedBid := testutil.MustParseDecimal(t, tt.expectedBid)
			expectedAsk := testutil.MustParseDecimal(t, tt.expectedAsk)

			testutil.AssertDecimalEqual(t, aggregated.Bid, expectedBid, "best bid")
			testutil.AssertDecimalEqual(t, aggregated.Ask, expectedAsk, "best ask")
		})
	}
}

package websocket

import (
	"bytes"
	"net/http"
	"net/http/httptest"
	"strings"
	"testing"
	"time"

	"github.com/gorilla/websocket"
)

func TestWebSocketHub_SingleClient(t *testing.T) {
	hub := NewHub()
	go hub.Run()
	defer hub.Stop()

	// Create test server
	server := httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		ServeWs(hub, w, r)
	}))
	defer server.Close()

	// Connect client
	wsURL := "ws" + strings.TrimPrefix(server.URL, "http")
	conn, _, err := websocket.DefaultDialer.Dial(wsURL, nil)
	if err != nil {
		t.Fatalf("dial failed: %v", err)
	}
	defer conn.Close()

	// Wait for client to register
	time.Sleep(50 * time.Millisecond)

	// Broadcast message
	testMsg := []byte(`{"type":"tick","symbol":"EURUSD","bid":1.0850,"ask":1.0852}`)
	hub.Broadcast(testMsg)

	// Verify client receives message
	conn.SetReadDeadline(time.Now().Add(time.Second))
	_, received, err := conn.ReadMessage()
	if err != nil {
		t.Fatalf("read failed: %v", err)
	}

	if !bytes.Equal(received, testMsg) {
		t.Errorf("got %s, want %s", received, testMsg)
	}
}

func TestWebSocketHub_MultipleClients(t *testing.T) {
	hub := NewHub()
	go hub.Run()
	defer hub.Stop()

	// Create test server
	server := httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		ServeWs(hub, w, r)
	}))
	defer server.Close()

	// Connect multiple clients
	wsURL := "ws" + strings.TrimPrefix(server.URL, "http")
	numClients := 5
	clients := make([]*websocket.Conn, numClients)

	for i := 0; i < numClients; i++ {
		conn, _, err := websocket.DefaultDialer.Dial(wsURL, nil)
		if err != nil {
			t.Fatalf("client %d dial failed: %v", i, err)
		}
		defer conn.Close()
		clients[i] = conn
	}

	// Wait for all clients to register
	time.Sleep(100 * time.Millisecond)

	// Broadcast message
	testMsg := []byte(`{"type":"tick","symbol":"BTCUSD","bid":95000,"ask":95010}`)
	hub.Broadcast(testMsg)

	// Verify all clients receive
	for i, client := range clients {
		client.SetReadDeadline(time.Now().Add(time.Second))
		_, received, err := client.ReadMessage()
		if err != nil {
			t.Fatalf("client %d read failed: %v", i, err)
		}
		if !bytes.Equal(received, testMsg) {
			t.Errorf("client %d: got %s, want %s", i, received, testMsg)
		}
	}
}

func TestWebSocketHub_ClientDisconnect(t *testing.T) {

func TestWebSocketHub_RaceCondition(t *testing.T) {
	// Run with: go test -race
	hub := NewHub()
	go hub.Run()
	defer hub.Stop()

	server := httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		ServeWs(hub, w, r)
	}))
	defer server.Close()

	// Connect and disconnect clients concurrently while broadcasting
	wsURL := "ws" + strings.TrimPrefix(server.URL, "http")
	done := make(chan bool)

	// Goroutine: connect/disconnect clients
	go func() {
		for i := 0; i < 10; i++ {
			conn, _, err := websocket.DefaultDialer.Dial(wsURL, nil)
			if err == nil {
				time.Sleep(10 * time.Millisecond)
				conn.Close()
			}
		}
		done <- true
	}()

	// Goroutine: broadcast messages
	go func() {
		for i := 0; i < 20; i++ {
			hub.Broadcast([]byte(`{"type":"tick"}`))
			time.Sleep(5 * time.Millisecond)
		}
		done <- true
	}()

	// Wait for both goroutines
	<-done
	<-done

	// If no race conditions detected by -race flag, test passes
}

package integration

import (
	"context"
	"net/http"
	"net/http/httptest"
	"strings"
	"testing"
	"time"

	"github.com/gorilla/websocket"
	"trading-engine/backend/lpmanager"
	"trading-engine/backend/websocket"
)

func TestLPToWebSocketFlow(t *testing.T) {
	// Integration test: LP quote → Manager → Hub → Client
	ctx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
	defer cancel()

	// Setup LP manager
	manager := lpmanager.NewLPManager()
	go manager.Run(ctx)

	// Setup WebSocket hub
	hub := websocket.NewHub()
	go hub.Run()
	defer hub.Stop()

	// Connect hub to LP manager quote stream
	go func() {
		for {
			select {
			case quote := <-manager.QuoteChannel():
				// Broadcast LP quote to WebSocket clients
				hub.Broadcast(quote.ToJSON())
			case <-ctx.Done():
				return
			}
		}
	}()

	// Create test WebSocket server
	server := httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		websocket.ServeWs(hub, w, r)
	}))
	defer server.Close()

	// Connect WebSocket client
	wsURL := "ws" + strings.TrimPrefix(server.URL, "http")
	conn, _, err := websocket.DefaultDialer.Dial(wsURL, nil)
	if err != nil {
		t.Fatalf("dial failed: %v", err)
	}
	defer conn.Close()

	// Wait for client registration
	time.Sleep(100 * time.Millisecond)

	// Simulate LP quote arriving
	mockQuote := lpmanager.Quote{
		Symbol: "EURUSD",
		Bid:    testutil.MustParseDecimal(t, "1.0850"),
		Ask:    testutil.MustParseDecimal(t, "1.0852"),
		LP:     "MockLP",
	}
	manager.ProcessQuote(mockQuote)

	// Verify client receives quote via WebSocket
	conn.SetReadDeadline(time.Now().Add(2 * time.Second))
	_, received, err := conn.ReadMessage()
	if err != nil {
		t.Fatalf("read failed: %v", err)
	}

	// Parse and verify quote
	var receivedQuote lpmanager.Quote
	err = json.Unmarshal(received, &receivedQuote)
	if err != nil {
		t.Fatalf("unmarshal failed: %v", err)
	}

	if receivedQuote.Symbol != "EURUSD" {
		t.Errorf("got symbol %s, want EURUSD", receivedQuote.Symbol)
	}
}
