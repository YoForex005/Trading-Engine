package bbook

import (

func TestAccountDeposit(t *testing.T) {
	tests := []struct {
		name            string
		initialBalance  string
		depositAmount   string
		expectedBalance string
		expectError     bool
	}{
		{
			name:            "standard deposit",
			initialBalance:  "1000.00",
			depositAmount:   "500.00",
			expectedBalance: "1500.00",
			expectError:     false,
		},
		{
			name:            "zero deposit rejected",
			initialBalance:  "1000.00",
			depositAmount:   "0.00",
			expectedBalance: "1000.00",
			expectError:     true,
		},
		{
			name:            "negative deposit rejected",
			initialBalance:  "1000.00",
			depositAmount:   "-100.00",
			expectedBalance: "1000.00",
			expectError:     true,
		},
		{
			name:            "large deposit",
			initialBalance:  "1000.00",
			depositAmount:   "1000000.00",
			expectedBalance: "1001000.00",
			expectError:     false,
		},
		{
			name:            "fractional cents",
			initialBalance:  "1000.00",
			depositAmount:   "100.125",
			expectedBalance: "1100.13", // Banker's rounding
			expectError:     false,
		},
	}

	for _, tt := range tests {
		tt := tt // Capture range variable
		t.Run(tt.name, func(t *testing.T) {
			t.Parallel()

			// Setup
			initialBal := testutil.MustParseDecimal(t, tt.initialBalance)
			depositAmt := testutil.MustParseDecimal(t, tt.depositAmount)
			expectedBal := testutil.MustParseDecimal(t, tt.expectedBalance)

			account := &Account{Balance: initialBal}

			// Execute
			err := account.Deposit(depositAmt)

			// Verify
			if tt.expectError {
				if err == nil {
					t.Error("expected error but got none")
				}
			} else {
				if err != nil {
					t.Fatalf("unexpected error: %v", err)
				}
				testutil.AssertDecimalEqual(t, account.Balance, expectedBal, "balance after deposit")
			}
		})
	}
}

package bbook

import (

func TestCalculatePositionProfit(t *testing.T) {
	tests := []struct {
		name          string
		symbol        string
		side          string // "buy" or "sell"
		lots          string
		entryPrice    string
		currentPrice  string
		expectedProfit string
	}{
		{
			name:          "buy position in profit",
			symbol:        "EURUSD",
			side:          "buy",
			lots:          "1.0",
			entryPrice:    "1.0850",
			currentPrice:  "1.0900",
			expectedProfit: "500.00", // 50 pips * $10/pip for 1 lot
		},
		{
			name:          "buy position in loss",
			symbol:        "EURUSD",
			side:          "buy",
			lots:          "1.0",
			entryPrice:    "1.0850",
			currentPrice:  "1.0800",
			expectedProfit: "-500.00", // -50 pips
		},
		{
			name:          "sell position in profit",
			symbol:        "EURUSD",
			side:          "sell",
			lots:          "1.0",
			entryPrice:    "1.0850",
			currentPrice:  "1.0800",
			expectedProfit: "500.00", // Price went down, profit for sell
		},
		{
			name:          "sell position in loss",
			symbol:        "EURUSD",
			side:          "sell",
			lots:          "1.0",
			entryPrice:    "1.0850",
			currentPrice:  "1.0900",
			expectedProfit: "-500.00",
		},
		{
			name:          "mini lot position",
			symbol:        "EURUSD",
			side:          "buy",
			lots:          "0.1",
			entryPrice:    "1.0850",
			currentPrice:  "1.0900",
			expectedProfit: "50.00", // 1/10 of standard lot
		},
		{
			name:          "micro lot position",
			symbol:        "EURUSD",
			side:          "buy",
			lots:          "0.01",
			entryPrice:    "1.0850",
			currentPrice:  "1.0900",
			expectedProfit: "5.00",
		},
		{
			name:          "no movement = no profit",
			symbol:        "EURUSD",
			side:          "buy",
			lots:          "1.0",
			entryPrice:    "1.0850",
			currentPrice:  "1.0850",
			expectedProfit: "0.00",
		},
	}

	for _, tt := range tests {
		tt := tt
		t.Run(tt.name, func(t *testing.T) {
			t.Parallel()

			lots := testutil.MustParseDecimal(t, tt.lots)
			entryPrice := testutil.MustParseDecimal(t, tt.entryPrice)
			currentPrice := testutil.MustParseDecimal(t, tt.currentPrice)
			expectedProfit := testutil.MustParseDecimal(t, tt.expectedProfit)

			position := &Position{
				Symbol:     tt.symbol,
				Side:       tt.side,
				Lots:       lots,
				EntryPrice: entryPrice,
			}

			profit := position.CalculateProfit(currentPrice)

			testutil.AssertDecimalEqual(t, profit, expectedProfit, "position profit")
		})
	}
}

	}

	for _, tt := range tests {

			t.Parallel()

			lots := testutil.MustParseDecimal(t, tt.lots)
			expectedMargin := testutil.MustParseDecimal(t, tt.expectedMargin)

			margin := CalculateMargin(tt.symbol, lots, tt.leverage)

	}
}

package bbook

import (
	"testing"

	"github.com/govalues/decimal"
	"trading-engine/backend/internal/testutil"
)

func TestOrderExecution_MarketOrder(t *testing.T) {
	tests := []struct {
		name              string
		orderType         string
		side              string
		lots              string
		accountBalance    string
		currentPrice      string
		expectedExecution bool
		expectedBalance   string
	}{
		{
			name:              "buy market order executes",
			orderType:         "market",
			side:              "buy",
			lots:              "1.0",
			accountBalance:    "10000.00",
			currentPrice:      "1.0850",
			expectedExecution: true,
			expectedBalance:   "9000.00", // 10000 - 1000 margin
		},
		{
			name:              "insufficient margin rejects order",
			orderType:         "market",
			side:              "buy",
			lots:              "1.0",
			accountBalance:    "500.00",
			currentPrice:      "1.0850",
			expectedExecution: false,
			expectedBalance:   "500.00",
		},
		{
			name:              "sell market order executes",
			orderType:         "market",
			side:              "sell",
			lots:              "1.0",
			accountBalance:    "10000.00",
			currentPrice:      "1.0850",
			expectedExecution: true,
			expectedBalance:   "9000.00",
		},
	}

	for _, tt := range tests {
		tt := tt
		t.Run(tt.name, func(t *testing.T) {
			t.Parallel()

			lots := testutil.MustParseDecimal(t, tt.lots)
			balance := testutil.MustParseDecimal(t, tt.accountBalance)
			price := testutil.MustParseDecimal(t, tt.currentPrice)
			expectedBal := testutil.MustParseDecimal(t, tt.expectedBalance)

			account := &Account{Balance: balance}
			order := &Order{
				Type: tt.orderType,
				Side: tt.side,
				Lots: lots,
			}

			executed := ExecuteMarketOrder(account, order, price)

			if executed != tt.expectedExecution {
				t.Errorf("execution: got %v, want %v", executed, tt.expectedExecution)
			}

			testutil.AssertDecimalEqual(t, account.Balance, expectedBal, "account balance after order")
		})
	}
}
