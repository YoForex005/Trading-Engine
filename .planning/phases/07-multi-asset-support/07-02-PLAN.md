// SessionPeriod represents a trading session within a day
type SessionPeriod struct {
    DayOfWeek   time.Weekday
    OpenHour    int
    OpenMinute  int
    CloseHour   int
    CloseMinute int
    SessionType string // "regular", "pre-market", "after-hours"
}

// TradingSession defines when a symbol can be traded
type TradingSession struct {
    ID        string          // Unique ID (e.g., "NYSE", "FOREX_24_5", "CRYPTO_24_7")
    Name      string          // Display name
    Timezone  string          // IANA timezone (e.g., "America/New_York")
    Sessions  []SessionPeriod // Multiple sessions per week
    Holidays  []time.Time     // Exchange-specific holidays
}

// TradingHoursManager manages all trading sessions
type TradingHoursManager struct {
    sessions map[string]*TradingSession
    mu       sync.RWMutex
}
```

**Functions:**
- NewTradingHoursManager() *TradingHoursManager - initializes with default sessions
- (m *TradingHoursManager) RegisterSession(session *TradingSession) - adds custom session
- (m *TradingHoursManager) IsMarketOpen(sessionID string, t time.Time) (bool, error) - checks if market open
- (m *TradingHoursManager) GetTradingHours(sessionID string) (*TradingSession, error) - retrieves session
- CreateDefaultSessions() map[string]*TradingSession - creates standard sessions

**Default Sessions:**
1. "FOREX_24_5" - Monday 00:00 UTC to Friday 23:59 UTC (5 days)
2. "CRYPTO_24_7" - Always open (7 days)
3. "NYSE" - Weekdays 9:30am-4:00pm ET (America/New_York timezone)
4. "NASDAQ" - Same as NYSE
5. "LSE" - Weekdays 8:00am-4:30pm GMT (Europe/London timezone)

CRITICAL timezone handling:
- Use time.LoadLocation() with IANA timezone identifiers
- Compare times in exchange's local timezone (not UTC conversion errors)
- Handle DST transitions automatically via timezone database

Reference RESEARCH.md "Pitfall 6: Session/Timezone Management Complexity" for edge cases.

package engine

import (
    "context"
    "fmt"
    "time"
    "github.com/epic1st/rtx/backend/lpmanager/metadata"
)

type OrderValidator struct {
    hoursManager *metadata.TradingHoursManager
    symbolRepo   SymbolRepository // Interface for getting symbol specs
}

func NewOrderValidator(hoursManager *metadata.TradingHoursManager, symbolRepo SymbolRepository) *OrderValidator {
    return &OrderValidator{
        hoursManager: hoursManager,
        symbolRepo:   symbolRepo,
    }
}

func (v *OrderValidator) ValidateOrder(ctx context.Context, order *Order) error {
    // 1. Get symbol spec
    spec, err := v.symbolRepo.GetBySymbol(ctx, order.Symbol)
    if err != nil {
        return fmt.Errorf("symbol not found: %w", err)
    }

    // 2. Check trading hours if symbol has TradingHoursID
    if spec.TradingHoursID != "" {
        isOpen, err := v.hoursManager.IsMarketOpen(spec.TradingHoursID, time.Now())
        if err != nil {
            return fmt.Errorf("failed to check trading hours: %w", err)
        }
        if !isOpen {
            return fmt.Errorf("market closed for %s (session: %s)", order.Symbol, spec.TradingHoursID)
        }
    }

    // 3. Validate tick size (order price must be multiple of tick size)
    // 4. Validate lot size (order volume must be >= min, <= max, multiple of step)
    // 5. Additional validations...

    return nil
}
```
