```go
package lpmanager

import (
    "fmt"
    "sync"
    "github.com/epic1st/rtx/backend/lpmanager/metadata"
)

// LPRouter routes orders and quotes to appropriate LP based on asset class
type LPRouter struct {
    routes map[metadata.AssetClass][]LPAdapter
    mu     sync.RWMutex
}

func NewLPRouter() *LPRouter {
    return &LPRouter{
        routes: make(map[metadata.AssetClass][]LPAdapter),
    }
}

// RegisterLP registers an LP for specific asset classes
func (r *LPRouter) RegisterLP(lp LPAdapter, assetClasses []metadata.AssetClass) {
    r.mu.Lock()
    defer r.mu.Unlock()

    for _, ac := range assetClasses {
        r.routes[ac] = append(r.routes[ac], lp)
    }
}

// RouteQuote gets quote from best LP for given symbol
func (r *LPRouter) RouteQuote(symbol string, assetClass metadata.AssetClass) (*Quote, error) {
    r.mu.RLock()
    adapters := r.routes[assetClass]
    r.mu.RUnlock()

    if len(adapters) == 0 {
        return nil, fmt.Errorf("no LP configured for asset class %s", assetClass)
    }

    // Try each LP in order until success (failover)
    var lastErr error
    for _, lp := range adapters {
        if !lp.IsConnected() {
            continue
        }

        // Get quote from this LP
        // NOTE: Actual implementation would call LP's GetQuote method
        // For now, placeholder returning first connected LP
        quote := Quote{
            Symbol: symbol,
            LP:     lp.ID(),
        }
        return &quote, nil
    }

    return nil, fmt.Errorf("all LPs failed for %s: %v", assetClass, lastErr)
}

// GetLPsForAssetClass returns registered LPs for asset class
func (r *LPRouter) GetLPsForAssetClass(ac metadata.AssetClass) []LPAdapter {
    r.mu.RLock()
    defer r.mu.RUnlock()
    return r.routes[ac]
}
```

**Default routing configuration:**
- AssetClassForex → [OANDA, InteractiveBrokers] (OANDA primary)

- AssetClassCommodity → [InteractiveBrokers] (IB only)
- AssetClassIndex → [InteractiveBrokers] (IB only)
- AssetClassCFD → [InteractiveBrokers, OANDA] (IB primary)

```go
package lpmanager

import (
    "testing"
    "github.com/stretchr/testify/assert"
    "github.com/stretchr/testify/require"

// Mock LP adapter for testing
type mockLPAdapter struct {
    id        string
    connected bool
}

func (m *mockLPAdapter) ID() string { return m.id }
func (m *mockLPAdapter) IsConnected() bool { return m.connected }
// ... other interface methods ...

func TestLPRouter_RegisterLP(t *testing.T) {
    router := NewLPRouter()
    mockLP := &mockLPAdapter{id: "test-lp", connected: true}

    router.RegisterLP(mockLP, []metadata.AssetClass{
        metadata.AssetClassStock,
        metadata.AssetClassForex,
    })

    stockLPs := router.GetLPsForAssetClass(metadata.AssetClassStock)
    forexLPs := router.GetLPsForAssetClass(metadata.AssetClassForex)

    assert.Len(t, stockLPs, 1)
    assert.Len(t, forexLPs, 1)
    assert.Equal(t, "test-lp", stockLPs[0].ID())
}

func TestLPRouter_RouteQuote_SelectsCorrectLP(t *testing.T) {
    router := NewLPRouter()

    oandaLP := &mockLPAdapter{id: "oanda", connected: true}
    binanceLP := &mockLPAdapter{id: "binance", connected: true}

    router.RegisterLP(oandaLP, []metadata.AssetClass{metadata.AssetClassForex})
    router.RegisterLP(binanceLP, []metadata.AssetClass{metadata.AssetClassCrypto})

    // Route forex symbol to OANDA
    forexQuote, err := router.RouteQuote("EURUSD", metadata.AssetClassForex)
    require.NoError(t, err)
    assert.Equal(t, "oanda", forexQuote.LP)

    // Route crypto symbol to Binance
    cryptoQuote, err := router.RouteQuote("BTCUSD", metadata.AssetClassCrypto)
    require.NoError(t, err)
    assert.Equal(t, "binance", cryptoQuote.LP)
}

func TestLPRouter_Failover_WhenPrimaryLPDisconnected(t *testing.T) {
    router := NewLPRouter()

    primaryLP := &mockLPAdapter{id: "primary", connected: false} // Disconnected
    backupLP := &mockLPAdapter{id: "backup", connected: true}

    router.RegisterLP(primaryLP, []metadata.AssetClass{metadata.AssetClassStock})
    router.RegisterLP(backupLP, []metadata.AssetClass{metadata.AssetClassStock})

    quote, err := router.RouteQuote("AAPL", metadata.AssetClassStock)
    require.NoError(t, err)
    assert.Equal(t, "backup", quote.LP, "Should failover to backup when primary disconnected")
}

func TestLPRouter_ErrorWhenNoLPForAssetClass(t *testing.T) {
    router := NewLPRouter()

    _, err := router.RouteQuote("UNKNOWN", metadata.AssetClassCommodity)
    assert.Error(t, err)
    assert.Contains(t, err.Error(), "no LP configured")
}
```
