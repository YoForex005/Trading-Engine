```go
package engine

import (
    "time"
    "github.com/govalues/decimal"
    "github.com/epic1st/rtx/backend/lpmanager/metadata"
)

type CFDCalculator struct{}

func NewCFDCalculator() *CFDCalculator {
    return &CFDCalculator{}
}

// CalculateMargin returns required margin for CFD position
// Formula: Position Value × Margin Rate
// Position Value = Volume × Contract Size × Price
func (c *CFDCalculator) CalculateMargin(
    volume decimal.Decimal,
    price decimal.Decimal,
    spec *metadata.SymbolSpec,
) decimal.Decimal {
    if spec.AssetClass != metadata.AssetClassCFD {
        return decimal.Zero // Not a CFD, no special margin
    }

    positionValue := volume.Mul(spec.ContractSize).Mul(price)
    requiredMargin := positionValue.Mul(spec.MarginRate)

    return requiredMargin
}

// CalculateSwap returns overnight financing charge/credit for CFD position
// Formula: Volume × Swap Rate × Multiplier
// Multiplier = 1 for most days, 3 on Wednesdays (weekend rollover)
func (c *CFDCalculator) CalculateSwap(
    volume decimal.Decimal,
    isLong bool,
    spec *metadata.SymbolSpec,
    date time.Time,
) decimal.Decimal {
    if spec.AssetClass != metadata.AssetClassCFD {
        return decimal.Zero // Not a CFD, no swap
    }

    // Select swap rate based on position direction
    swapRate := spec.SwapRateLong
    if !isLong {
        swapRate = spec.SwapRateShort
    }

    // Base swap calculation
    baseSwap := volume.Mul(swapRate)

    // Triple swap on Wednesdays (accounts for weekend)
    multiplier := decimal.NewFromInt(1)
    if date.Weekday() == time.Wednesday {
        multiplier = decimal.NewFromInt(3)
    }

    return baseSwap.Mul(multiplier)
}

// IsWednesday checks if given date is Wednesday (for triple swap)
func IsWednesday(t time.Time) bool {
    return t.Weekday() == time.Wednesday
}
```

CRITICAL:
- Use decimal.Decimal for all calculations (no float64)
- Triple swap ONLY on Wednesdays
- Swap can be positive (credit) or negative (charge) depending on interest rates
- Reference RESEARCH.md "Pattern 3: CFD Swap/Rollover Calculator" for formula details

```go
// ApplyDailySwaps applies overnight financing charges to all open CFD positions
// Should be called once per day at market rollover time (typically 10pm UTC / 5pm ET)
func (e *Engine) ApplyDailySwaps() error {
    e.mu.Lock()
    defer e.mu.Unlock()

    now := time.Now()
    rolloverTime := time.Date(now.Year(), now.Month(), now.Day(), 22, 0, 0, 0, time.UTC)

    if now.Before(rolloverTime) {
        return nil // Not yet rollover time
    }

    for _, account := range e.accounts {
        for _, position := range account.Positions {
            // Get symbol spec
            spec, err := e.symbolRepo.GetBySymbol(context.Background(), position.Symbol)
            if err != nil || spec.AssetClass != metadata.AssetClassCFD {
                continue // Skip non-CFDs
            }

            // Calculate swap
            isLong := position.Side == "buy"
            swap := e.cfdCalc.CalculateSwap(position.Volume, isLong, spec, now)

            // Apply swap to account balance
            account.Balance = account.Balance.Add(swap) // Can be positive or negative
            account.Equity = account.Equity.Add(swap)

            // Log swap application
            log.Printf("[CFD Swap] Account %s, Position %s: %s swap %s",
                account.ID, position.Symbol, position.Side, swap)
        }
    }

    return nil
}
```

```go
package engine

import (

func TestCFDCalculator_CalculateMargin(t *testing.T) {
    calc := NewCFDCalculator()
    spec := &metadata.SymbolSpec{
        AssetClass:   metadata.AssetClassCFD,
        ContractSize: decimal.NewFromInt(100),
        MarginRate:   decimal.MustParse("0.10"), // 10% margin
    }

    volume := decimal.MustParse("10.0") // 10 lots
    price := decimal.MustParse("50.00")

    // Expected: 10 lots × 100 contract size × $50 price × 0.10 margin = $5,000
    margin := calc.CalculateMargin(volume, price, spec)
    expected := decimal.MustParse("5000.00")

    assert.True(t, margin.Equal(expected), "Margin should be $5,000, got %s", margin)
}

func TestCFDCalculator_CalculateSwap_Regular(t *testing.T) {
    calc := NewCFDCalculator()
    spec := &metadata.SymbolSpec{
        AssetClass:    metadata.AssetClassCFD,
        SwapRateLong:  decimal.MustParse("0.50"), // $0.50 per lot
        SwapRateShort: decimal.MustParse("-0.30"), // -$0.30 per lot
    }

    volume := decimal.MustParse("10.0")
    monday := time.Date(2026, 1, 19, 22, 0, 0, 0, time.UTC) // Monday

    // Long position
    swapLong := calc.CalculateSwap(volume, true, spec, monday)
    expectedLong := decimal.MustParse("5.00") // 10 lots × $0.50 × 1
    assert.True(t, swapLong.Equal(expectedLong))

    // Short position
    swapShort := calc.CalculateSwap(volume, false, spec, monday)
    expectedShort := decimal.MustParse("-3.00") // 10 lots × -$0.30 × 1
    assert.True(t, swapShort.Equal(expectedShort))
}

func TestCFDCalculator_CalculateSwap_Wednesday_TripleSwap(t *testing.T) {
```
