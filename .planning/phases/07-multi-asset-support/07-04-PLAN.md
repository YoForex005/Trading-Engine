            },
        }
        json.NewEncoder(w).Encode(mockResponse)
    }))
    defer server.Close()

    adapter := NewBinanceAdapter()
    adapter.baseURL = server.URL
    symbols, err := adapter.GetSymbols()

    require.NoError(t, err)
    assert.GreaterOrEqual(t, len(symbols), 3, "Should have at least BTC, SOL, XRP")

    // Verify all have AssetClassCrypto

        assert.Equal(t, "Binance", sym.LPSource)
        assert.Equal(t, "CRYPTO_24_7", sym.TradingHoursID)
    }
}
